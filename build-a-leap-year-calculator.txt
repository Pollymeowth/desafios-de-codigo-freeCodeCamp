function isLeapYear(year){
 let isLapYear = ((year % 4 === 0 && year % 100 !== 0) || (year % 400 === 0))
  ? `${year} is a leap year.`
  :`${year} is not a leap year.`;

  return isLapYear;
}

let year = 2024;
let result = isLeapYear(year)
console.log(result);
